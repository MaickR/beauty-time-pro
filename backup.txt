// Funciones auxiliares de tiempo y fechas locales
const timeToMins = (timeStr) => {
  if (!timeStr) return null;
  const [h, m] = timeStr.split(':').map(Number);
  return h * 60 + m;
};
const minsToTime = (mins) => {
  const h = Math.floor(mins / 60).toString().padStart(2, '0');
  const m = (mins % 60).toString().padStart(2, '0');
  return `${h}:${m}`;
};
const getLocalISODate = (date) => {
  if(!date) return '';
  const offset = date.getTimezoneOffset();
  const localDate = new Date(date.getTime() - (offset * 60 * 1000));
  return localDate.toISOString().split('T')[0];
};

// Formateador de moneda
const formatMoney = (amount, currency = 'MXN') => {
  return new Intl.NumberFormat('es-MX', { style: 'currency', currency: currency }).format(amount || 0);
};

// Calculador Inteligente de Suscripciones
const getSubscriptionStatus = (studio) => {
  if (!studio || !studio.subscriptionStart) return null;
  const today = new Date();
  today.setHours(0,0,0,0);
  
  let currentDueStr = studio.paidUntil || studio.subscriptionStart;
  let dueParts = currentDueStr.split('-');
  if(dueParts.length < 3) return null;
  
  let dueDate = new Date(parseInt(dueParts[0]), parseInt(dueParts[1])-1, parseInt(dueParts[2]));
  
  const diffTime = dueDate.getTime() - today.getTime();
  const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  
  return {
      cutDay: parseInt(dueParts[2]),
      dueDateStr: dueDate.toLocaleDateString('es-ES', { day: 'numeric', month: 'long', year: 'numeric' }),
      daysRemaining: diffDays,
      status: diffDays < 0 ? 'OVERDUE' : (diffDays <= 5 ? 'WARNING' : 'ACTIVE')
  };
};

      const staffBookings = bookings.filter(b => b.studioId === currentStudio.id && b.staffId === selectedStaff && b.date === dateStr && b.status !== 'cancelled');
      
      const staffMember = currentStudio.staff.find(s => s.id === selectedStaff);
      
      const studioStartMins = timeToMins(sched.openTime);
      const studioEndMins = timeToMins(sched.closeTime);
      
      const shiftStartMins = staffMember?.shiftStart ? timeToMins(staffMember.shiftStart) : studioStartMins;
      const shiftEndMins = staffMember?.shiftEnd ? timeToMins(staffMember.shiftEnd) : studioEndMins;
      
      const effectiveStart = Math.max(studioStartMins, shiftStartMins);
      const effectiveEnd = Math.min(studioEndMins, shiftEndMins);

      const breakStartMins = staffMember?.breakStart ? timeToMins(staffMember.breakStart) : null;
      const breakEndMins = staffMember?.breakEnd ? timeToMins(staffMember.breakEnd) : null;
      const hasBreak = breakStartMins !== null && breakEndMins !== null && breakStartMins < breakEndMins;

      const slots = [];

      for (let time = effectiveStart; time < effectiveEnd; time += 30) {
        const slotEnd = time + totalDuration;
        const timeStr = minsToTime(time);
        let status = 'AVAILABLE';
        
        const isLunchTime = hasBreak && time >= breakStartMins && time < breakEndMins;
        const serviceOverlapsLunch = hasBreak && time < breakStartMins && slotEnd > breakStartMins;

        if (isLunchTime) {
          status = 'BREAK_TIME';

          status = 'TOO_SHORT';

            if (time < bEnd && slotEnd > bStart) {
              status = 'OCCUPIED';
              break;
            }
          }
        }
        
        if (dateStr === getLocalISODate(new Date())) {
          const nowMins = new Date().getHours() * 60 + new Date().getMinutes();
          if (time <= nowMins) continue;
        }
        
        if (status === 'TOO_SHORT') continue; 
        
        slots.push({ time: timeStr, status });
      }
      return slots;
    };

    const smartSlots = getSmartSlots();
    const availableSlotsCount = smartSlots.filter(s => s.status === 'AVAILABLE').length;
    
    const selectedDateStr = getLocalISODate(selectedDate);
    const isDayHoliday = currentStudio.holidays?.includes(selectedDateStr);
    const isDayClosed = !currentStudio.schedule[DAYS_OF_WEEK[selectedDate.getDay()]]?.isOpen;

    return (
      <div className="min-h-screen bg-white flex flex-col font-sans">
        
        {/* MODAL DE CONFIRMACIÓN */}
        {bookingSuccess && (
          <div className="fixed inset-0 bg-slate-900/90 backdrop-blur-sm z-[100] flex items-center justify-center p-4">
            <div className="bg-white rounded-[3rem] p-10 max-w-lg w-full text-center shadow-2xl">
              <div className="w-24 h-24 bg-green-100 text-green-500 rounded-full flex items-center justify-center mx-auto mb-6 shadow-inner">
                <CheckCircle2 className="w-12 h-12" />
              </div>
              <h2 className="text-3xl font-black italic uppercase tracking-tighter mb-4 text-slate-800">¡Cita Confirmada!</h2>
              <p className="text-lg font-medium text-slate-600 mb-8">
                Muchas gracias por tu preferencia, <strong className="text-pink-600 uppercase">{bookedClientName}</strong>. Tu especialista te estará esperando.
              </p>
              
              <div className="bg-orange-50 border border-orange-200 rounded-3xl p-6 mb-8 text-left shadow-sm">
                <div className="flex items-start gap-4">
                  <div className="bg-orange-100 p-2 rounded-xl text-orange-600 shrink-0">
                    <Clock className="w-6 h-6" />
                  </div>
                  <div>
                    <h4 className="font-black text-orange-800 uppercase text-xs tracking-widest mb-1">Aviso Importante</h4>
                    <p className="text-orange-900 text-sm font-medium leading-relaxed">
                      Tu reservación cuenta con una tolerancia máxima de <strong>15 minutos</strong>. Si no te presentas en ese tiempo, tu lugar será asignado a otro cliente para no afectar la agenda.
                    </p>
                  </div>
                </div>
              </div>

              <button 
                onClick={() => setBookingSuccess(false)} 
                className="w-full bg-slate-900 text-white font-black py-5 rounded-[2rem] hover:bg-black transition-all uppercase tracking-widest text-sm shadow-xl active:scale-95"
              >
                Entendido, Gracias
              </button>
            </div>
          </div>
        )}

        <header className="p-6 md:p-8 border-b border-slate-100 flex justify-between items-center sticky top-0 bg-white/90 backdrop-blur-md z-50 shadow-sm">
          <div className="flex items-center gap-4">
            <div className="bg-slate-950 p-3 rounded-2xl"><UserCheck className="text-pink-500" /></div>
            <div>
              <h1 className="text-xl md:text-2xl font-black italic tracking-tighter uppercase leading-none">{currentStudio.name}</h1>
              <p className="text-[10px] font-black text-slate-400 uppercase tracking-widest mt-1">Reserva de Citas en Línea</p>
            </div>
          </div>
          <button onClick={logout} className="p-3 bg-slate-50 rounded-full hover:bg-slate-100 text-slate-400"><LogOut /></button>
        </header>

        <main className="max-w-4xl mx-auto w-full p-4 md:p-8 space-y-8 pb-32">
          
          <section className="bg-slate-900 rounded-[3rem] p-8 md:p-10 text-white shadow-2xl">
            <h3 className="text-xl md:text-2xl font-black italic uppercase mb-8 flex items-center gap-3 text-pink-500"><span className="bg-white/10 w-8 h-8 rounded-full flex items-center justify-center text-sm">1</span> Elige tu Especialista</h3>
            <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-4">
              {activeStaff.map(st => (
                <div 
                  key={st.id} 
                  onClick={() => handleStaffSelect(st.id)} 
                  className={`cursor-pointer p-6 rounded-3xl border-2 transition-all ${selectedStaff === st.id ? 'bg-pink-600 border-pink-400 shadow-xl shadow-pink-900/50 scale-[1.02]' : 'bg-slate-800 border-slate-700 hover:border-slate-500'}`}
                >
                  <h4 className="text-xl font-black uppercase mb-3 text-white">{st.name}</h4>
                  <div className="flex flex-wrap gap-1.5">
                    {st.specialties.map(spec => (
                      <span key={spec} className={`px-2 py-1 rounded-lg text-[8px] font-black uppercase border ${selectedStaff === st.id ? 'bg-pink-500 border-pink-400 text-white' : 'bg-slate-900 border-slate-700 text-slate-400'}`}>
                        {spec}
                      </span>
                    ))}
                  </div>
                </div>
              ))}
              {activeStaff.length === 0 && <p className="col-span-full text-center text-red-400 font-bold">No hay especialistas disponibles hoy.</p>}
            </div>
          </section>

          <div className={`transition-all duration-700 ${selectedStaff ? 'opacity-100 translate-y-0' : 'opacity-50 pointer-events-none -translate-y-4 hidden'}`}>
            <section className="bg-slate-50 rounded-[3rem] p-8 md:p-10 border border-slate-200 shadow-sm">
              <h3 className="text-xl md:text-2xl font-black italic uppercase tracking-tighter mb-8 text-slate-800 flex items-center gap-3">
                <span className="bg-pink-100 text-pink-600 w-8 h-8 rounded-full flex items-center justify-center text-sm">2</span> Servicios Disponibles
              </h3>
              <p className="text-xs font-bold text-slate-500 uppercase mb-4">Servicios que realiza {currentStudio.staff?.find(s=>s.id === selectedStaff)?.name || 'este especialista'}:</p>
              
              <div className="flex flex-col gap-3">
                {availableServices.map((s, i) => (
                  <button key={i} onClick={() => toggleClientService(s)} className={`text-left p-5 rounded-2xl border transition-all flex justify-between items-center gap-4 ${clientSelectedServices.find(x=>x.name===s.name) ? 'bg-pink-600 border-pink-600 text-white shadow-lg scale-[1.01]' : 'bg-white border-slate-200 hover:border-pink-300'}`}>
                    <div className="flex flex-col">
                      <span className={`font-black text-sm uppercase ${clientSelectedServices.find(x=>x.name===s.name) ? 'text-white' : 'text-slate-800'}`}>
                        {clientSelectedServices.find(x=>x.name===s.name) && '✓ '} {s.name}
                      </span>
                      <span className={`text-[10px] font-bold mt-1 ${clientSelectedServices.find(x=>x.name===s.name) ? 'text-pink-200' : 'text-slate-400'}`}><Clock className="w-3 h-3 inline"/> {s.duration} MINUTOS</span>
                    </div>
                    {s.price > 0 && (
                      <span className={`font-black text-lg ${clientSelectedServices.find(x=>x.name===s.name) ? 'text-white' : 'text-green-600'}`}>{formatMoney(s.price, currency)}</span>
                    )}
                  </button>
                ))}
                {availableServices.length === 0 && selectedStaff && <p className="text-slate-400 italic font-bold">El especialista no tiene tratamientos habilitados en este momento.</p>}
              </div>

              {clientSelectedServices.length > 0 && (
                <div className="mt-8 bg-slate-900 text-white p-6 rounded-3xl flex justify-between items-center shadow-xl">
                  <div>
                    <p className="text-[10px] font-black text-pink-400 uppercase tracking-widest mb-1">Total a Pagar en Sucursal</p>
                    <p className="text-3xl font-black text-green-400">{formatMoney(totalPrice, currency)}</p>
                  </div>
                  <div className="text-right">
                    <p className="text-[10px] font-black text-slate-400 uppercase tracking-widest mb-1">Tiempo Estimado</p>
                    <p className="text-xl font-black">{totalDuration} min</p>
                  </div>
                </div>
              )}
            </section>
          </div>

          <div className={`transition-all duration-700 delay-100 ${clientSelectedServices.length > 0 ? 'opacity-100 translate-y-0' : 'opacity-50 pointer-events-none -translate-y-4 hidden'}`}>
            <section className="bg-white rounded-[3rem] border border-slate-200 shadow-lg overflow-hidden">
              <div className="bg-slate-50 p-8 md:p-10 border-b border-slate-100 text-center">
                 <h3 className="text-xl md:text-2xl font-black italic uppercase tracking-tighter text-slate-800 flex items-center justify-center gap-3">
                    <span className="bg-pink-100 text-pink-600 w-8 h-8 rounded-full flex items-center justify-center text-sm">3</span> Día y Hora
                 </h3>
                 <p className="text-[10px] font-black text-slate-500 uppercase mt-4 tracking-widest bg-white inline-block px-4 py-2 rounded-xl shadow-sm border border-slate-200">
                    Duración Estimada: <span className="text-pink-600 text-sm">{totalDuration} min</span>
                 </p>
              </div>
              
              <div className="p-6 md:p-10">
                <div className="flex items-center justify-between mb-8">
                  <button onClick={() => changeMonth(-1)} className="p-3 bg-slate-50 hover:bg-slate-100 rounded-full transition-colors"><ChevronLeft className="w-6 h-6 text-slate-600"/></button>
                  <h4 className="font-black uppercase tracking-widest text-lg md:text-xl text-slate-800">{selectedDate.toLocaleString('es-ES', { month: 'long', year: 'numeric' })}</h4>
                  <button onClick={() => changeMonth(1)} className="p-3 bg-slate-50 hover:bg-slate-100 rounded-full transition-colors"><ChevronRight className="w-6 h-6 text-slate-600"/></button>
                </div>
                
                <div className="grid grid-cols-7 gap-2 md:gap-3">
                  {['DOM','LUN','MAR','MIE','JUE','VIE','SAB'].map((d, index) => <div key={index} className="text-center text-[10px] md:text-xs font-black text-slate-400 uppercase tracking-widest mb-2">{d}</div>)}
                  {calendarDays.map((day, i) => {
                    const dateObj = day ? new Date(selectedDate.getFullYear(), selectedDate.getMonth(), day) : null;
                    const dateStr = dateObj ? getLocalISODate(dateObj) : null;
                    const isSelected = dateStr && dateStr === getLocalISODate(selectedDate);
                    
                    const isHol = dateStr && currentStudio.holidays?.includes(dateStr);
                    const isOp = dateObj && currentStudio.schedule[DAYS_OF_WEEK[dateObj.getDay()]]?.isOpen;
                    const isPst = dateObj && dateObj < new Date(new Date().setHours(0,0,0,0));
                    const isDisabled = isHol || !isOp || isPst;

                    return (
                      <div key={i} className="flex items-center justify-center">
                        {day && (
                          <button 
                            disabled={isDisabled}
                            onClick={() => {setSelectedDate(dateObj); setSelectedTime('');}} 
                            className={`w-full h-12 md:h-16 rounded-2xl font-black text-lg md:text-xl transition-all flex items-center justify-center border-2 
                              ${isDisabled ? 'opacity-30 cursor-not-allowed bg-slate-100 border-slate-200 text-slate-400' : 
                                isSelected ? 'bg-slate-900 border-slate-900 text-white shadow-xl scale-110 z-10' : 
                                'bg-white border-slate-100 text-slate-700 hover:border-pink-300 hover:text-pink-600'}
                            `}
                          >
                            {day}
                          </button>
                        )}
                      </div>
                    );
                  })}
                </div>
              </div>

              {/* AVISOS CONDICIONALES PARA EL DÍA SELECCIONADO */}
              {isDayHoliday ? (
                <div className="p-8 bg-red-50 border-t border-red-100 text-center">
                  <AlertTriangle className="w-8 h-8 text-red-400 mx-auto mb-3"/>
                  <p className="text-sm font-black text-red-800 uppercase tracking-widest">Día Festivo / Inhábil</p>
                  <p className="text-xs text-red-600 mt-2">El Studio permanecerá cerrado este día.</p>
                </div>
              ) : isDayClosed ? (
                <div className="p-8 bg-slate-100 border-t border-slate-200 text-center">
                  <Clock className="w-8 h-8 text-slate-400 mx-auto mb-3"/>
                  <p className="text-sm font-black text-slate-600 uppercase tracking-widest">Día de Descanso</p>
                  <p className="text-xs text-slate-500 mt-2">El Studio no labora los días {DAYS_OF_WEEK[selectedDate.getDay()]}.</p>
                </div>
              ) : (
                <div className="p-6 md:p-10 bg-slate-50 border-t border-slate-100">
                  <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4 mb-8">
                    <h4 className="text-sm font-black uppercase text-slate-800 flex items-center gap-2 tracking-widest"><Clock className="w-5 h-5 text-pink-600"/> Horarios Disponibles</h4>
                    <div className="flex flex-wrap items-center gap-2 text-[10px] font-black uppercase tracking-widest bg-white px-4 py-2 rounded-xl shadow-sm border border-slate-200">
                      <span className="flex items-center gap-1 text-green-600"><div className="w-2 h-2 bg-green-500 rounded-full shadow-sm"></div> Libre</span>
                      <span className="flex items-center gap-1 text-red-500"><div className="w-2 h-2 bg-red-500 rounded-full shadow-sm"></div> Ocupado</span>
                      <span className="flex items-center gap-1 text-yellow-600 ml-2"><Coffee className="w-3 h-3"/> Comida</span>
                    </div>
                  </div>

                  <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 gap-3 md:gap-4 max-h-80 overflow-y-auto custom-scrollbar pr-2">
                    {smartSlots.length > 0 ? smartSlots.map(slot => (
                      <button key={slot.time} disabled={slot.status !== 'AVAILABLE'} onClick={() => setSelectedTime(slot.time)} 
                        className={`py-4 md:py-5 rounded-2xl font-black text-sm md:text-base transition-all border-2 shadow-sm flex justify-center items-center gap-2
                          ${slot.status === 'AVAILABLE' ? (selectedTime === slot.time ? 'bg-green-600 border-green-500 text-white scale-105 shadow-xl' : 'bg-white border-green-200 text-green-700 hover:bg-green-50 hover:border-green-400') : ''}
                          ${slot.status === 'OCCUPIED' ? 'bg-red-50 border-red-200 text-red-400 cursor-not-allowed opacity-80 line-through' : ''}
                          ${slot.status === 'BREAK_TIME' ? 'bg-yellow-50 border-yellow-200 text-yellow-600 cursor-not-allowed' : ''}
                        `}
                      >
                        {slot.status === 'AVAILABLE' && <span className="w-2 h-2 rounded-full bg-green-500 shadow-sm"></span>}
                        {slot.status === 'OCCUPIED' && <span className="w-2 h-2 rounded-full bg-red-500 shadow-sm"></span>}
                        {slot.status === 'BREAK_TIME' && <Coffee className="w-4 h-4"/>}
                        {slot.time}
                      </button>
                    )) : null}
                  </div>

                  {smartSlots.length > 0 && availableSlotsCount === 0 && (
                    <div className="mt-6 bg-red-50 border-2 border-red-200 p-5 rounded-2xl flex items-start gap-4">
                      <AlertTriangle className="w-8 h-8 text-red-500 shrink-0"/>
                      <div>
                        <p className="text-sm font-black text-red-800 uppercase tracking-widest mb-1">Agenda Llena / Tiempo Insuficiente</p>
                        <p className="text-xs text-red-700 font-bold leading-relaxed">
                          El tiempo requerido para tus tratamientos ({totalDuration} min) excede la disponibilidad actual de este especialista para el día de hoy. Por favor, selecciona otro día en el calendario.
                        </p>
                      </div>
                    </div>
                  )}

                  {smartSlots.length === 0 && (
                    <p className="text-center py-10 text-sm text-slate-500 font-bold uppercase tracking-widest bg-white rounded-3xl border border-slate-200">El especialista no trabaja en este horario.</p>
                  )}
                </div>
              )}
            </section>
          </div>

          <div className={`transition-all duration-700 delay-200 ${selectedTime ? 'opacity-100 translate-y-0' : 'opacity-50 pointer-events-none -translate-y-4 hidden'}`}>
            <section className="bg-slate-900 rounded-[3.5rem] p-10 shadow-2xl flex flex-col text-white">
              <h3 className="text-xl md:text-2xl font-black italic uppercase tracking-tighter mb-8 flex items-center gap-3">
                 <span className="bg-pink-600 text-white w-8 h-8 rounded-full flex items-center justify-center text-sm">4</span> Tus Datos de Contacto
              </h3>
              
              {/* PREGUNTAS DE COLORACIÓN (Solo si eligió tinte, balayage, etc) */}
              {requiresColorInfo && (
                <div className="col-span-full bg-pink-950/40 p-6 rounded-3xl border border-pink-900/50 mb-8">
                  <p className="text-[10px] font-black text-pink-400 uppercase tracking-widest mb-4 flex items-center gap-2"><Palette className="w-4 h-4"/> Detalles de tu Coloración (Opcional)</p>
                  <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                    <div>
                      <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-2 ml-2">Marca de Tinte Preferida</label>
                      <input type="text" className="w-full px-5 py-3 bg-slate-800 border border-slate-700 rounded-xl font-bold text-white outline-none focus:border-pink-500 transition-all" placeholder="Ej: L'Oréal, Wella..." value={colorBrand} onChange={e => setColorBrand(e.target.value)} />
                    </div>
                    <div>
                      <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-2 ml-2">Número / Tono del Tinte</label>
                      <input type="text" className="w-full px-5 py-3 bg-slate-800 border border-slate-700 rounded-xl font-bold text-white outline-none focus:border-pink-500 transition-all" placeholder="Ej: 7.1, 8.0..." value={colorNumber} onChange={e => setColorNumber(e.target.value)} />
                    </div>
                  </div>
                </div>
              )}

              <div className="space-y-6 mb-10">
                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                  <div>
                    <label className="block text-[10px] font-black text-pink-400 uppercase tracking-widest mb-3 ml-2">Tu Nombre Completo <span className="text-red-500 text-sm">*</span></label>
                    <input type="text" className="w-full px-6 py-4 bg-slate-800 border border-slate-700 rounded-2xl font-bold text-white outline-none focus:border-pink-500 transition-all" placeholder="Ej: María López" value={clientName} onChange={e => setClientName(e.target.value)} required />
                  </div>
                  <div>
                    <label className="block text-[10px] font-black text-pink-400 uppercase tracking-widest mb-3 ml-2">Teléfono Celular <span className="text-red-500 text-sm">*</span></label>
                    <input type="tel" className="w-full px-6 py-4 bg-slate-800 border border-slate-700 rounded-2xl font-bold text-white outline-none focus:border-pink-500 transition-all" placeholder="Ej: 55 1234 5678" value={clientPhone} onChange={e => setClientPhone(e.target.value)} required />
                  </div>
                </div>
                {currentStudio.branches.length > 1 && (
                  <div>
                    <label className="block text-[10px] font-black text-pink-400 uppercase tracking-widest mb-3 ml-2">Elige tu Sucursal <span className="text-red-500 text-sm">*</span></label>
                    <select className="w-full px-6 py-4 bg-slate-800 border border-slate-700 rounded-2xl font-bold text-white outline-none focus:border-pink-500 appearance-none" value={selectedBranch} onChange={e => setSelectedBranch(e.target.value)}>
                      {currentStudio.branches.map((b, i) => <option key={i} value={b}>{b}</option>)}
                    </select>
                  </div>
                )}
              </div>

              <button onClick={handleClientBooking} disabled={!clientName.trim() || !clientPhone.trim() || !selectedStaff || !selectedTime} className="w-full bg-pink-600 text-white font-black py-6 md:py-8 rounded-[2.5rem] shadow-xl shadow-pink-900/50 hover:bg-pink-500 transition-all uppercase tracking-widest text-sm md:text-base disabled:opacity-50 disabled:cursor-not-allowed active:scale-95">
                Confirmar mi Reservación
              </button>
            </section>
          </div>
        </main>
      </div>
    );
  }

  return null;
}
